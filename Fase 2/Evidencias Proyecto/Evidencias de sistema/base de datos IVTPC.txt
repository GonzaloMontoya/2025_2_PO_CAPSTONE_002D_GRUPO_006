-- Tabla de Usuarios (Simulando tu sistema actual)
create table users (
  id text primary key,
  "nombreCompleto" text,
  email text,
  telefono text,
  rol text,
  estado text,
  password text
);

-- Tabla de Vehículos
create table vehicles (
  id text primary key,
  patente text,
  marca text,
  modelo text,
  chofer text,
  "choferId" text,
  llegada timestamptz
);

-- Tabla de Citas
create table appointments (
  id text primary key,
  marca text,
  modelo text,
  patente text,
  chofer text,
  fecha timestamptz,
  motivo text,
  "estadoReparacion" text,
  "fechaFinalizacionEstimada" timestamptz,
  "customStatus" text,
  imagenes text[], -- Array de strings para URLs
  "comentariosMecanico" text
);

-- Tabla de Reportes
create table maintenance_reports (
  id text primary key,
  "reportDate" timestamptz,
  "vehicleModel" text,
  "vehicleBrand" text,
  "vehicleYear" text,
  "vehicleLicensePlate" text,
  "driverName" text,
  "scheduledAppointment" timestamptz,
  "maintenanceStart" timestamptz,
  "maintenanceEnd" timestamptz,
  "preMaintenanceComments" text,
  "workDoneComments" text,
  "mechanicNotes" text,
  "beforeImage" text,
  "afterImage" text
);

-- Tabla de Notificaciones
create table notifications (
  id text primary key,
  "userId" text,
  "messageKey" text,
  read boolean,
  timestamp timestamptz
);

-- Tabla de Documentos
create table technical_documents (
  id text primary key,
  name text,
  "uploadDate" timestamptz,
  "fileType" text,
  "fileData" text, -- Guardaremos la URL o Base64 aquí
  "fileSize" numeric
);
